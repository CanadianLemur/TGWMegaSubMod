	increase_agrarian_tensions = { ##Repression of Zapatistas
		fire_only_once = no
		allowed = {
			tag = MEX
			check_variable = {
				var = MEX.communist_tensions_amount
				value = 1
				compare = less_than
			}
		}
		available = {
			always = yes
		}
		is_good = yes
		
		days_remove = 20
		days_re_enable = 0

			custom_effect_tooltip = MSMMexico_increase_agrarian_tensions

			if={
				limit={
					check_variable = {

				}
				add_to_variable = {
					var = MEX.communist_appeasement_amount

				}
			}
		}
	}
