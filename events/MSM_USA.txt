####Choose Your Candidate - Republican Vote Question####
country_event = {
	id =  MSMUSA.1
	title = MSMUSA.1.t
	desc = MSMUSA.1.d
	picture = GFX_report_event_generic_read_write
	
	fire_only_once = yes
	is_triggered_only = no
	
	mean_time_to_happen = {
        days = 10
    }
	
	trigger = {
		date > 1912.3.1
	}

	option = {	#Progressives & Conservatives happy - Taft runs again and wins 
		name = MSMUSA.1.a
		ai_chance = { factor = 100 }
		trigger = { 
			AND = {
				MSM_USA_progressives_happy = yes 
				MSM_USA_conservatives_happy = yes 
			}
		}
		effect = { set_variable = { USA_candidate = 0 } }
	}
	option = {	#Conservatives happy - Teddy runs against Taft and Wilson wins
		name = MSMUSA.1.b
		ai_chance = { factor = 100 }
		trigger = { 
			AND = {
				MSM_USA_conservatives_happy = yes 
				NOT = { MSM_USA_progressives_happy = yes }
				USA = { has_country_flag = MSM_USA_teddys_angry }
			}
		}
		effect = { set_variable = { USA_candidate = 1 } }
	}
	option = {	#Conservatives & teddy happy - Debs runs and grabs the progressive and centrist votes for the W
		name = MSMUSA.1.c
		ai_chance = { factor = 100 }
		trigger = { 
			AND = {
				MSM_USA_conservatives_happy = yes 
				NOT = { MSM_USA_progressives_happy = yes }
				NOT = { USA = { has_country_flag = MSM_USA_teddys_angry } }
			}
		}
		effect = { set_variable = { USA_candidate = 2 } }
	}
	option = {	#Everyone's pissed - Teddy runs and wins the primary, beats Wilson
		name = MSMUSA.1.d
		ai_chance = { factor = 100 }
		trigger = { 
			AND = {
				NOT = { MSM_USA_conservatives_happy = yes }
				NOT = { MSM_USA_progressives_happy = yes }
			}
		}
		effect = { set_variable = { USA_candidate = 3 } }
	}
}

# Election (1913)
#____________________________________________________________
country_event = {
	id = MSMUSA.30
	title = MSMUSA.30.t
	desc = MSMUSA.30.d

    mean_time_to_happen = {
        days = 0
    }

    fire_only_once = yes

    trigger = {
    	tag = USA
        date > 1912.11.1
        date < 1913.10.10
        USA = {
	        has_government = democratic
	        not = {
	            has_country_flag = new_1913_president_usa
	        }
        }
    }
	hidden_effect = {
			if = {
				limit = {
					OR = {
						has_completed_focus = USA_voter_registration_act
						has_completed_focus = USA_union_representation_act
					}
				}
				country_event = { id = usa.32 days = 730 } # midterms
			}
			else = {
				country_event = { id = usa.35 days = 730 } # midterms
			}
		}
	# Wilson
	option = {
		name = MSMUSA.30.a
		ai_chance = { factor = 100 }
		set_country_flag = new_1913_president_usa
		trigger = { check_variable = { USA_candidate = 0 } }
		hidden_effect = {
			news_event = { days = 150 id = MSMUSA.31 }
		}
	}
	# Teddy
	option = {
		name = MSMUSA.30.b
		ai_chance = { factor = 100 }
		set_country_flag = new_1913_president_usa
		trigger = { check_variable = { USA_candidate = 1 } }
		hidden_effect = {
			news_event = { days = 150 id = MSMUSA.32 }
		}
	}
	# Taft
	option = {
		name = MSMUSA.30.c
		ai_chance = { factor = 100 }
		set_country_flag = new_1913_president_usa
		trigger = { check_variable = { USA_candidate = 2 } }
		hidden_effect = {
			news_event = { days = 150 id = MSMUSA.33 }
		}
	}
	# Debs
	option = {
		name = MSMUSA.30.d
		ai_chance = { factor = 100 }
		set_country_flag = new_1913_president_usa
		trigger = { check_variable = { USA_candidate = 3 } }
		hidden_effect = {
			news_event = { days = 150 id = MSMUSA.34 }
		}
	}
}
