	immediate = {
		hidden_effect = {
		
			QIN = {
				set_equipment_fraction = 0.2
				delete_unit_template_and_units = { division_template = "Qing Brigade" }
				delete_unit_template_and_units = { division_template = "Qing Cavalry" }
				load_oob = "QIN_1911_Revolution"
			}
			
		
			CHI = {
					set_cosmetic_tag = CHI_REV
					set_politics = { ruling_party = neutrality }
					set_capital = 620
					transfer_state = 325
					transfer_state = 599
					transfer_state = 594
					transfer_state = 592
					transfer_state = 620
					transfer_state = 606
					transfer_state = 749
					transfer_state = 595
					transfer_state = 621
					transfer_state = 609
					transfer_state = 610

				set_stability = 0.2
				add_manpower = -2200000
				
				load_oob = "PRC_1911_Wuchang"
				
				add_timed_idea = {
				idea = hoi4tgw_idea_revolution
				days = 250
				}
				
			}
		
			release = MON # Release Mongolia
			release = TIB # Release Tibet
			release = TAN # Release Tanu
			RUS = {
				puppet = TAN
			}
		}
		
	}
	
	
	# Become China
	option = {
		name = MSMWuchang.2.a

		hidden_effect = {
			set_global_flag = xinhai_revolution
				CHI = {
				if =  {
					limit = {
						QIN = {
							has_tech = infantry_equipment_1
						}
					}
					CHI = {
						set_technology = {
							infantry_equipment_1 = 1
						}
						add_equipment_to_stockpile = {
							type = infantry_equipment_1
							amount = 50000
							producer = QIN
						}
					}
				}
				if =  {
					limit = {
						not = {
						QIN = {
							has_tech = infantry_equipment_1
						}
						}
					}
					CHI = {
						set_technology = {
							infantry_equipment_1 = 0
						}
						add_equipment_to_stockpile = {
							type = infantry_equipment_0
							amount = 50000
							producer = QIN
						}
					}
				}
				change_tag_from = QIN
				add_stability = 0.80
				add_war_support = 0.85
				
			}
			
			
			QIN = {
				set_war_support = 0.40
				set_stability = 0.2
			}
			
			
			
			
		}
		custom_effect_tooltip = roc_chosen
		
		QIN = {
				drop_cosmetic_tag = yes
			}
			CHI = {
				declare_war_on = {
					target = QIN
					type = civil_war
				}
			
				create_country_leader = {
					name = "Sun Yat-sen"
					desc = "POLITICS_SUN_YAT_SEN_DESC"
					picture = "gfx/leaders/Sun_Yat_sen.dds"
					expire = "1925.1.1"
					ideology = despotism
					traits = { }
				}
				
			}
		
		IF = {
			limit = {
				has_global_flag = qing_support
			}
			add_war_support = -0.2
			add_stability = -0.1
		}
		
		IF = {
			limit = {
				has_global_flag = rebels_support
			}
			add_war_support = 0.2
			set_stability = 0.85
			add_political_power = 145
		}
		
	}
	
	# Support Qing
	option = {
		name = MSMWuchang.2.b
		trigger = {
			is_ai = no
		}
		hidden_effect = {
			set_global_flag = xinhai_revolution

			set_stability = 0.9
			load_oob = "QIN_Civil_War"

			add_popularity = {
				ideology = neutrality
				popularity = 100
			}
		}
		custom_effect_tooltip = qing_chosen
	}
	
	
}
